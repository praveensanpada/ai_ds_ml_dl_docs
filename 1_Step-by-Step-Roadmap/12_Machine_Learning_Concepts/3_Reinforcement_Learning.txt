import numpy as np

# Q-table: rows = states, columns = actions
Q = np.zeros((5, 2))

# Simple example of Q-update (Q-learning formula)
state = 0
action = 1
reward = 10
next_state = 1
alpha = 0.1
gamma = 0.9

Q[state, action] = Q[state, action] + alpha * (reward + gamma * np.max(Q[next_state]) - Q[state, action])
print("Updated Q-Table:\n", Q)
